**1. `app/src/lib/sync/backoff.ts` (NEU):**

```typescript
// Exponential backoff with full jitter (AWS-style).
// Research §3: 2^attempts * base, cap at 60s, random ∈ [0, cap].
// attempts counter wird vom SyncWorker erhöht, nicht hier.

const BASE_MS = 1000;   // 1 Sekunde erste Verzögerung
const CAP_MS = 60_000;  // 60s-Cap

export function nextBackoffMs(attempts: number, now: () => number = () => Math.random()): number {
  if (attempts <= 0) return 0;
  const exp = Math.min(CAP_MS, BASE_MS * Math.pow(2, attempts - 1));
  return Math.floor(now() * exp);
}

export const MAX_ATTEMPTS = 10;
```

```typescript
// Leichter Event-Channel (kein externes lib — React Native hat kein eingebautes
// EventEmitter-Modul, das hier nötig wäre).
// UI (Plan 03-06) subscribed via useSyncEvents hook.
// Sentry-Capture wird in SyncWorker direkt gemacht.

import type { OutboxEntry } from '@spatenstich/shared';
import { ConflictError } from '../errors';

export type SyncEvent =
  | { type: 'push_start'; entry: OutboxEntry }
  | { type: 'push_success'; entry: OutboxEntry }
  | { type: 'push_retry'; entry: OutboxEntry; nextAttempt: number; nextDelayMs: number }
  | { type: 'push_conflict'; entry: OutboxEntry; error: ConflictError }
  | { type: 'push_permanent_failure'; entry: OutboxEntry; lastError: string }
  | { type: 'pull_start'; entity: string }
  | { type: 'pull_success'; entity: string; rowsFetched: number; serverNow: string }
  | { type: 'pull_failure'; entity: string; error: string }
  | { type: 'status_change'; status: 'idle' | 'syncing' | 'offline' | 'degraded' };

type Listener = (event: SyncEvent) => void;

const listeners = new Set<Listener>();

export const syncEvents = {
  on(listener: Listener): () => void {
    listeners.add(listener);
    return () => listeners.delete(listener);
  },
  emit(event: SyncEvent): void {
    for (const l of listeners) {
      try {
        l(event);
      } catch (e) {
        if (__DEV__) console.error('[syncEvents] listener threw', e);
      }
    }
  },
  // Test-utility: clear listeners between tests
  _reset(): void {
    listeners.clear();
  },
};
```

```typescript
import * as Sentry from '@sentry/react-native';
import { storage } from '../../storage';
import { supabase } from '../supabase';
import { ConflictError } from '../errors';
import { useAuthStore } from '../../stores/authStore';
import { syncEvents } from './events';
import { nextBackoffMs, MAX_ATTEMPTS } from './backoff';
import type {
  EntityName,
  OutboxEntry,
  GardenRow,
  GardenMemberRow,
  ProfileRow,
  VereinsregelnRow,
  InviteCodeRow,
  PhotoQueueRow,
} from '@spatenstich/shared';
import {
  gardenFromDb,
  profileFromDb,
  vereinsregelnFromDbRows,
  gardenMemberFromDb,
  inviteCodeFromDb,
  photoQueueFromDb,
  vereinsregelnToDbRows,
} from '../mappers/rowMappers';

// Entities, die gepullt werden (photo_queue ist nur-lokal; Plan 03-05 uploaded selbst)
const PULL_ENTITIES: EntityName[] = [
  'gardens',
  'garden_members',
  'profiles',
  'vereinsregeln',
  'invite_codes',
];

// ── Push-Lock ───────────────────────────────────────────────────────────────
let pushInFlight = false;

// ── Public API ──────────────────────────────────────────────────────────────

/**
 * Push alle pending Outbox-Einträge sequentiell.
 * Idempotent bei parallelem Aufruf (zweiter Aufruf ist No-Op wenn erster noch läuft).
 */
export async function push(): Promise<void> {
  if (pushInFlight) return;
  pushInFlight = true;
  syncEvents.emit({ type: 'status_change', status: 'syncing' });
  try {
    const entries = await storage.listOutboxEntries(50); // Batch-Limit
    for (const entry of entries) {
      await pushOne(entry);
    }
  } finally {
    pushInFlight = false;
    const remaining = await storage.listOutboxEntries(1);
    syncEvents.emit({
      type: 'status_change',
      status: remaining.length > 0 ? 'degraded' : 'idle',
    });
  }
}

/**
 * Delta-Pull für eine Entity.
 * Nutzt sync_state.lastPullAt als `updated_at > X`-Filter.
 * Aktualisiert sync_state auf server_now() nach Erfolg.
 */
export async function pull(entity: EntityName): Promise<void> {
  syncEvents.emit({ type: 'pull_start', entity });
  try {
    const state = await storage.getSyncState(entity);
    const lastPullAt = state?.lastPullAt;

    // server_now() als neuer Checkpoint
    const { data: nowData, error: nowErr } = await supabase.rpc('server_now');
    if (nowErr) throw nowErr;
    const serverNow = nowData as string;

    const rowsFetched = await pullEntity(entity, lastPullAt);

    await storage.setSyncState({
      entity,
      lastPullAt: serverNow,
      lastPushAt: state?.lastPushAt ?? null,
    });
    syncEvents.emit({ type: 'pull_success', entity, rowsFetched, serverNow });
  } catch (e) {
    const msg = e instanceof Error ? e.message : String(e);
    Sentry.captureException(e, { tags: { sync_phase: 'pull', entity } });
    syncEvents.emit({ type: 'pull_failure', entity, error: msg });
    throw e;
  }
}

/**
 * Voller Sync: alle Entities pullen + Outbox pushen.
 * Wird beim Reconnect + App-Start (nach auth) aufgerufen.
 */
export async function syncAll(): Promise<void> {
  for (const entity of PULL_ENTITIES) {
    try {
      await pull(entity);
    } catch (e) {
      if (__DEV__) console.warn(`[syncAll] pull ${entity} failed`, e);
      // Continue mit nächster Entity — partial-sync ist erlaubt
    }
  }
  await push();
}

/** Test-utility: reset internal state between tests. */
export function _resetSyncWorker(): void {
  pushInFlight = false;
}

// ── Private: Push Handlers ─────────────────────────────────────────────────

async function pushOne(entry: OutboxEntry): Promise<void> {
  syncEvents.emit({ type: 'push_start', entry });
  try {
    await dispatchPush(entry);
    // Success → aus Outbox entfernen
    await storage.deleteOutboxEntry(entry.id);
    syncEvents.emit({ type: 'push_success', entry });
  } catch (e) {
    await handlePushError(entry, e);
  }
}

async function dispatchPush(entry: OutboxEntry): Promise<void> {
  switch (entry.entity) {
    case 'gardens':     return pushGarden(entry);
    case 'profiles':    return pushProfile(entry);
    case 'vereinsregeln': return pushVereinsregeln(entry);
    case 'garden_members': return pushGardenMember(entry);
    case 'invite_codes': return pushInviteCode(entry);
    case 'photo_queue':  return pushPhotoQueue(entry);
    default:
      throw new Error(`Unknown entity for push: ${entry.entity}`);
  }
}

async function pushGarden(entry: OutboxEntry): Promise<void> {
  const row = entry.payload as GardenRow & { plz?: string | null; klimazone?: string | null; archetype?: string | null };
  const userId = useAuthStore.getState().userId;
  if (!userId) throw new Error('no_user');
  if (entry.operation === 'delete') {
    // Soft-delete via UPDATE deleted_at (D-16 Soft-Delete-Konvention)
    const { error } = await supabase
      .from('gardens')
      .update({
        deleted_at: new Date().toISOString(),
        updated_at: row.updatedAt,
        updated_by_user_id: userId,
      })
      .eq('id', entry.rowId);
    if (error) throw error;
    return;
  }
  const { error } = await supabase.from('gardens').upsert(
    {
      id: row.id,
      name: row.name,
      plz: row.plz ?? null,
      klimazone: row.klimazone ?? null,
      archetype: row.archetype ?? null,
      updated_at: row.updatedAt,
      updated_by_user_id: userId,
      deleted_at: row.deletedAt,

    { onConflict: 'id' },

  if (error) throw error;
}

  // Server-side: Upsert der neuen + DELETE der nicht mehr enthaltenen.
  const row = entry.payload as VereinsregelnRow;

  if (!userId) throw new Error('no_user');
  const dbRows = vereinsregelnToDbRows(row, userId);
  if (dbRows.length === 0) {
    // Alle Regeln gelöscht → leere Tabelle für diesen garden_id
    const { error } = await supabase
      .from('vereinsregeln')
      .delete()
      .eq('garden_id', row.gardenId);
    if (error) throw error;
    return;
  }
  // Upsert alle aktuellen Regeln
  const { error: upsertErr } = await supabase
    .from('vereinsregeln')
    .upsert(dbRows, { onConflict: 'id' });
  if (upsertErr) throw upsertErr;
  // Stale-IDs löschen (nicht mehr in der lokalen Row)
  const currentIds = dbRows.map((r) => r.id);
  const { error: deleteErr } = await supabase
    .from('vereinsregeln')
    .delete()
    .eq('garden_id', row.gardenId)
    .not('id', 'in', `(${currentIds.map((id) => `'${id}'`).join(',')})`);
  if (deleteErr) throw deleteErr;
}

// ── Private: Error Handling ─────────────────────────────────────────────────

async function handlePushError(entry: OutboxEntry, e: unknown): Promise<void> {
  const err = e as { code?: string; message?: string };
  const code = err.code;
  const msg = err.message ?? String(e);

  // P9011 (LWW-Reject): Server hat älteren Write verweigert → ConflictError, aus Queue entfernen
  if (code === 'P9011') {
    const conflictError = new ConflictError(entry.entity, entry.rowId, e);
    await storage.deleteOutboxEntry(entry.id);
    syncEvents.emit({ type: 'push_conflict', entry, error: conflictError });
    Sentry.captureMessage('sync: LWW conflict discarded', {
      level: 'warning',
      tags: { entity: entry.entity, row_id: entry.rowId },
    });
    return;
  }

  // P9010 (missing updated_at) — Client-Bug, Sentry + drop
  if (code === 'P9010') {
    await storage.deleteOutboxEntry(entry.id);
    Sentry.captureException(e, {
      tags: { sync_phase: 'push', entity: entry.entity, error_kind: 'missing_updated_at' },
    });
    syncEvents.emit({ type: 'push_permanent_failure', entry, lastError: 'P9010 missing updated_at' });
    return;
  }

  const nextAttempts = entry.attempts + 1;

  // Max-Attempts erreicht → permanent_failure, bleibt in Queue für User-Entscheidung (Plan 03-06)
  if (nextAttempts >= MAX_ATTEMPTS) {
    await storage.updateOutboxEntry(entry.id, { attempts: nextAttempts, lastError: msg });
    syncEvents.emit({ type: 'push_permanent_failure', entry: { ...entry, attempts: nextAttempts, lastError: msg }, lastError: msg });
    Sentry.captureException(e, {
      tags: { sync_phase: 'push', entity: entry.entity, error_kind: 'max_attempts' },
    });
    return;
  }

  // Retry mit Backoff
  const delayMs = nextBackoffMs(nextAttempts);
  await storage.updateOutboxEntry(entry.id, { attempts: nextAttempts, lastError: msg });
  syncEvents.emit({ type: 'push_retry', entry: { ...entry, attempts: nextAttempts, lastError: msg }, nextAttempt: nextAttempts, nextDelayMs: delayMs });
  // NOTE: Der eigentliche Retry passiert beim nächsten push()-Call (Trigger-driven).
  // Wir speichern hier nur, dass der Entry attempts=N hat. SyncTriggers.ts ruft push()
  // periodisch auf (Pull-All) und nach Events.
}

// ── Private: Pull Handlers ─────────────────────────────────────────────────

async function pullEntity(entity: EntityName, lastPullAt: string | null): Promise<number> {
  const userId = useAuthStore.getState().userId;
  const activeGardenId = useAuthStore.getState().activeGardenId;
  if (!userId) throw new Error('no_user');

  switch (entity) {
    case 'gardens': {
      // Gardens des Users (via RLS): Alle Gardens, in denen der User Member ist.
      // Keine garden_id-Filter nötig — RLS filtert serverseitig.
      let q = supabase.from('gardens').select('*');
      if (lastPullAt) q = q.gt('updated_at', lastPullAt);
      const { data, error } = await q;
      if (error) throw error;
      const rows = (data ?? []).map(gardenFromDb);
      if (rows.length > 0) await storage.upsertRowsFromServer('gardens', rows);
      return rows.length;
    }
    case 'profiles': {
      // Profile des Users selbst + aller Members der aktiven Gärten
      let q = supabase
        .from('profiles')
        .select('id, display_name, created_at, updated_at, updated_by_user_id, deleted_at, locale');
      if (lastPullAt) q = q.gt('updated_at', lastPullAt);
      const { data, error } = await q;
      if (error) throw error;
      const rows = (data ?? []).map(profileFromDb);
      if (rows.length > 0) await storage.upsertRowsFromServer('profiles', rows);
      return rows.length;
    }
    case 'vereinsregeln': {
      if (!activeGardenId) return 0;
      let q = supabase.from('vereinsregeln').select('*').eq('garden_id', activeGardenId);
      if (lastPullAt) q = q.gt('updated_at', lastPullAt);
      const { data, error } = await q;
      if (error) throw error;
      // Wenn eine oder mehrere neue Regel-Rows kommen: lokal aggregierte Row neu schreiben.
      // Wir pullen IMMER ALLE Regeln des Gardens (nicht nur die geänderten), damit die
      // Aggregation stimmt. Delta-Pull ist hier "gibt es Änderungen?"-Check.
      if ((data ?? []).length === 0) return 0;
      const { data: allData, error: allErr } = await supabase
        .from('vereinsregeln')
        .select('*')
        .eq('garden_id', activeGardenId);
      if (allErr) throw allErr;
      const aggregated = vereinsregelnFromDbRows(allData ?? [], activeGardenId);
      if (aggregated) await storage.upsertRowFromServer('vereinsregeln', aggregated);
      return (allData ?? []).length;
    }
    case 'garden_members': {
      if (!activeGardenId) return 0;
      // garden_members hat kein updated_at — pullen IMMER alle des aktiven Gartens.
      const { data, error } = await supabase
        .from('garden_members')
        .select('garden_id, user_id, role, joined_at')
        .eq('garden_id', activeGardenId);
      if (error) throw error;
      const rows = (data ?? []).map(gardenMemberFromDb);
      await storage.upsertRowsFromServer('garden_members', rows);
      return rows.length;
    }
    case 'invite_codes': {
      if (!activeGardenId) return 0;
      const { data, error } = await supabase
        .from('invite_codes')
        .select('*')
        .eq('garden_id', activeGardenId);
      if (error) throw error;
      const rows = (data ?? []).map(inviteCodeFromDb);
      await storage.upsertRowsFromServer('invite_codes', rows);
      return rows.length;
    }
    default:
      return 0;
  }
}
```

**1. `app/src/lib/sync/SyncTriggers.ts` (NEU):**

```typescript
// Subscribt auf 3 Event-Quellen und ruft den SyncWorker:
//   - NetInfo offline→online: syncAll()
//   - AppState background→active: syncAll()
//   - scheduleWriteDebounced(): push() nach 500ms debounce
// Ergibt eine unregister-Funktion, die alles aufräumt.
import NetInfo, { type NetInfoSubscription } from '@react-native-community/netinfo';
import { AppState, type AppStateStatus } from 'react-native';
import { syncAll, push } from './SyncWorker';

const WRITE_DEBOUNCE_MS = 500;

let writeDebounceTimer: ReturnType<typeof setTimeout> | null = null;

/** Debounced push() — 500ms nach letztem Write. */
export function scheduleWriteDebounced(): void {
  if (writeDebounceTimer) clearTimeout(writeDebounceTimer);
  writeDebounceTimer = setTimeout(() => {
    writeDebounceTimer = null;
    push().catch((e) => {
      if (__DEV__) console.warn('[SyncTriggers] debounced push failed', e);
    });
  }, WRITE_DEBOUNCE_MS);
}

/**
 * Registriert NetInfo + AppState Subscriptions.
 * Gibt unregister-Funktion zurück (call bei unmount / logout).
 */
export function registerSyncTriggers(): () => void {
  // ── NetInfo: offline → online ────────────────────────────────────────────
  let wasOffline = false;
  const netInfoUnsub: NetInfoSubscription = NetInfo.addEventListener((state) => {
    const isConnected = state.isConnected === true && state.isInternetReachable !== false;
    if (!isConnected) {
      wasOffline = true;
      return;
    }
    if (wasOffline && isConnected) {
      wasOffline = false;
      syncAll().catch((e) => {
        if (__DEV__) console.warn('[SyncTriggers] reconnect syncAll failed', e);
      });
    }
  });

  // ── AppState: background → active ────────────────────────────────────────
  let lastState: AppStateStatus = AppState.currentState;
  const appStateUnsub = AppState.addEventListener('change', (state) => {
    if (lastState !== 'active' && state === 'active') {
      syncAll().catch((e) => {
        if (__DEV__) console.warn('[SyncTriggers] foreground syncAll failed', e);
      });
    }
    lastState = state;
  });

  return () => {
    netInfoUnsub();
    appStateUnsub.remove();
    if (writeDebounceTimer) {
      clearTimeout(writeDebounceTimer);
      writeDebounceTimer = null;
    }
  };
}
```

```typescript
import { syncAll, _resetSyncWorker } from '../SyncWorker';
import { syncEvents } from '../events';
import { storage as storageA } from '../../../storage';

// Mock-Setup: in-memory supabase-shim teilt state zwischen beiden "Devices"
jest.mock('../../supabase', () => {
  const sharedState: Record<string, any[]> = { gardens: [] };
  let clock = 1_000_000; // monotonic

  const api = {
    rpc: jest.fn((name) => {
      if (name === 'server_now') {
        clock += 100;
        return Promise.resolve({ data: new Date(clock).toISOString(), error: null });
      }
      return Promise.resolve({ data: null, error: null });
    }),
    from: jest.fn((table: string) => ({
      select: jest.fn().mockReturnValue({
        gt: jest.fn().mockImplementation((_col, ts) =>
          Promise.resolve({ data: sharedState[table].filter((r) => r.updated_at > ts), error: null }),
        ),
        eq: jest.fn().mockReturnValue({
          maybeSingle: jest.fn().mockImplementation(() =>
            Promise.resolve({ data: sharedState[table][0] ?? null, error: null }),
          ),
        }),
      }),
      upsert: jest.fn().mockImplementation((row: any) => {
        // LWW simulieren: wenn existing.updated_at > row.updated_at → P9011

        if (existing && existing.updated_at >= row.updated_at) {
          return Promise.resolve({ data: null, error: { code: 'P9011', message: 'older write' } });
        }
        sharedState[table] = sharedState[table].filter((r) => r.id !== row.id).concat(row);
        return Promise.resolve({ data: row, error: null });
      }),
    })),
    _sharedState: sharedState,
    _advanceClock: (ms: number) => { clock += ms; },
  };
  return { supabase: api };
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| SyncWorker → Supabase | Authenticated REST/RPC calls — Auth via anon key + JWT header |
| SyncWorker → storage | Local DB — keine externen Angreifer, nur Prozess-internal |
| NetInfo / AppState Events | Gerät-internal, keine Trust-Boundary |
| syncEvents (EventChannel) | In-process, keine externen Subscribers |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-04-01 | Spoofing | Fremder User könnte synthetische Outbox-Einträge mit anderen userIds injizieren | mitigate | `pushGarden/pushProfile/pushVereinsregeln` stempeln `updated_by_user_id` mit `authStore.userId` (Pattern 6). Server-LWW-Trigger setzt `updated_by_user_id` erneut aus `auth.uid()` — clientseitige Manipulation wird überschrieben. |
| T-3-04-02 | Tampering | Angreifer könnte Outbox-Payload manipulieren, um fremde row_id zu pushen | mitigate | RLS (Phase 02.5) verwirft `UPDATE` für nicht-member-Rows serverseitig (42501). Lokaler Angriff ist harmlos — kein Fremd-Write möglich. |
| T-3-04-03 | Information Disclosure | Delta-Pull könnte Rows anderer Gärten pullen | accept | RLS filtert serverseitig via `is_garden_member`. Pull-Ergebnis enthält nur Rows, die der User sehen darf. Sentry-Breadcrumbs enthalten keine Row-Payloads, nur IDs. |
| T-3-04-04 | Denial of Service | Infinite Retry-Loop bei permanentem Server-Fehler | mitigate | `MAX_ATTEMPTS=10` + exponential backoff mit 60s-cap. Nach 10 Versuchen wird Eintrag auf 'permanent_failure' gekennzeichnet — kein weiteres Retry ohne User-Eingriff. |
| T-3-04-05 | Information Disclosure | Sentry-Logs könnten sensitive Row-Payloads enthalten | mitigate | Breadcrumbs enthalten nur {entity, row_id, operation, attempts} — KEINE Payload-Daten. `captureException` extras enthalten nur IDs. Sentry-Scrubber kann zusätzlich konfiguriert werden (Phase 4). |
| T-3-04-06 | Tampering | Race-Condition: push während Outbox-Write → lost update | mitigate | `pushInFlight`-Boolean serialisiert parallele `push()`-Calls. SqliteAdapter nutzt EXCLUSIVE-Lock für `listOutboxEntries + deleteOutboxEntry`-Sequenzen. IndexedDbAdapter serialisiert via Transaction-Queue. |
| T-3-04-07 | Denial of Service | Böser Push könnte Server überlasten bei Dauerreconnect | mitigate | Single-User-Context (D-13) → realistisch max. 2 Devices pro Garden.
