    ```ts
    import { useEffect, useRef, useState } from 'react';
    import NetInfo from '@react-native-community/netinfo';
    import { getStorage } from '@/lib/storage';
    import { syncEvents, type SyncEvent } from '@/lib/sync/events';
    import type { OutboxEntry } from '@shared/types/storage';

    export type SyncStatusValue = 'synced' | 'syncing' | 'degraded' | 'offline';

    export interface SyncStatus {
      status: SyncStatusValue;
      pendingCount: number;
      failedCount: number;
      /** Interne Flag, ob gerade aktiv gepusht/gepullt wird (aus status_change-Event). */
      activelySyncing: boolean;
    }

    const DEBOUNCE_MS = 300;

    export function useSyncStatus(): SyncStatus {
      const [pendingCount, setPendingCount] = useState(0);
      const [failedCount, setFailedCount] = useState(0);
      const [activelySyncing, setActivelySyncing] = useState(false);
      const [isConnected, setIsConnected] = useState<boolean>(true);
      const debounceTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

      // NetInfo subscription
      useEffect(() => {
        const unsub = NetInfo.addEventListener((state) => {
          setIsConnected(state.isConnected === true);
        });
        // Initial-Fetch
        NetInfo.fetch().then((state) => setIsConnected(state.isConnected === true));
        return () => unsub();
      }, []);

      const refreshCounts = () => {
        if (debounceTimerRef.current) clearTimeout(debounceTimerRef.current);
        debounceTimerRef.current = setTimeout(async () => {
          try {
            const entries = await getStorage().listOutboxEntries();
            const pending = entries.filter((e: OutboxEntry) => e.status !== 'failed').length;
            const failed = entries.filter((e: OutboxEntry) => e.status === 'failed').length;
            setPendingCount(pending);
            setFailedCount(failed);
          } catch {
            // Adapter-Fehler still schlucken — UI zeigt dann nur status_change
          }
        }, DEBOUNCE_MS);
      };

      useEffect(() => {
        // Initial-Fetch ohne Debounce
        (async () => {
          try {
            const entries = await getStorage().listOutboxEntries();
            setPendingCount(entries.filter((e: OutboxEntry) => e.status !== 'failed').length);
            setFailedCount(entries.filter((e: OutboxEntry) => e.status === 'failed').length);
          } catch {
            /* noop */
          }
        })();

        const unsub = syncEvents.on((evt: SyncEvent) => {
          if (evt.type === 'status_change') {
            setActivelySyncing(evt.status !== 'idle');
          }
          // Jedes Event triggert Re-Count (debounced)
          refreshCounts();
        });
        return () => {
          unsub();
          if (debounceTimerRef.current) clearTimeout(debounceTimerRef.current);
        };
      }, []);

      let status: SyncStatusValue;
      if (!isConnected) {
        status = 'offline';
      } else if (failedCount > 0) {
        status = 'degraded';
      } else if (activelySyncing || pendingCount > 0) {
        status = 'syncing';
      } else {
        status = 'synced';
      }

      return { status, pendingCount, failedCount, activelySyncing };
    }
    ```

    ```ts
    import { SyncWorker, PULL_ENTITIES } from '../SyncWorker';
    import { syncEvents, type SyncEvent } from '../events';
    import { createInMemoryAdapter } from '../../storage/__tests__/helpers/inMemoryAdapter';
    import { createSupabaseShim } from './helpers/supabaseShim';

    describe('SyncWorker.retryOp / discardOp / pullAll (Klassen-API)', () => {
      afterEach(() => {
        syncEvents._reset();
      });

      async function seedFailedOutboxEntry(storage: ReturnType<typeof createInMemoryAdapter>) {
        // Seed: ein failed-Entry in der Outbox, plus zugehörige Row (damit Delta-Pull
        // Sinn macht — die lokale Row kann dann durch den Server-Stand überschrieben werden)
        await storage.writeWithOutbox(
          'gardens',
          {
            id: 'g-1', name: 'Lokal',
            updated_at: '2026-04-01T10:00:00.000Z',
            owner_user_id: 'u-dirk', deleted_at: null,
          },
          { entity: 'gardens', rowId: 'g-1', op: 'upsert', payload: { id: 'g-1' } },
        );
        const [entry] = await storage.listOutboxEntries();
        // Markieren als 'failed' mit hohen attempts
        await storage.updateOutboxEntry(entry.id, {
          status: 'failed',
          attempts: 10,
          lastError: '500 Server Error',
        });
        return entry.id;
      }

      test('Test 1: retryOp setzt attempts=0, clear lastError, status=pending, triggert syncAll', async () => {
        const storage = createInMemoryAdapter();
        const supabase = createSupabaseShim();
        const worker = new SyncWorker({ storage, supabase: supabase.client });
        const opId = await seedFailedOutboxEntry(storage);

        const events: SyncEvent[] = [];
        const unsub = syncEvents.on((e) => events.push(e));
        await worker.retryOp(opId);
        unsub();

        const after = (await storage.listOutboxEntries()).find((e) => e.id === opId);
        expect(after?.attempts).toBe(0);
        expect(after?.lastError).toBeNull();
        // status kann nach syncAll bereits mutiert sein (pending → in_flight → failed/gone),
        // daher assertion: entweder noch pending, oder Entry wurde erfolgreich abgearbeitet (removed)
        expect(after === undefined || after.status !== 'failed').toBe(true);
        // status_change:syncing wurde emittiert
        expect(events.some((e) => e.type === 'status_change' && e.status === 'syncing')).toBe(true);
      });

      test('Test 2: retryOp mit nicht-existenter ID wirft "outbox_entry_not_found"', async () => {
        const storage = createInMemoryAdapter();
        const supabase = createSupabaseShim();
        const worker = new SyncWorker({ storage, supabase: supabase.client });

        await expect(worker.retryOp('nonexistent-id')).rejects.toThrow('outbox_entry_not_found');
      });

      test('Test 3: discardOp löscht Outbox-Entry', async () => {
        const storage = createInMemoryAdapter();
        const supabase = createSupabaseShim();
        const worker = new SyncWorker({ storage, supabase: supabase.client });
        const opId = await seedFailedOutboxEntry(storage);

        await worker.discardOp(opId);

        const after = await storage.listOutboxEntries();
        expect(after.find((e) => e.id === opId)).toBeUndefined();
      });

      test('Test 4: discardOp triggert Delta-Pull für betroffenes Entity', async () => {
        const storage = createInMemoryAdapter();
        const supabase = createSupabaseShim();
        const fromSpy = jest.spyOn(supabase.client, 'from');
        const worker = new SyncWorker({ storage, supabase: supabase.client });
        const opId = await seedFailedOutboxEntry(storage);
        fromSpy.mockClear();

        await worker.discardOp(opId);

        // Erwarte mindestens einen .from('gardens')-Call für den Delta-Pull
        expect(fromSpy).toHaveBeenCalledWith('gardens');
      });

      test('Test 5: discardOp mit nicht-existenter ID ist idempotent (no throw)', async () => {
        const storage = createInMemoryAdapter();
        const supabase = createSupabaseShim();
        const worker = new SyncWorker({ storage, supabase: supabase.client });

        await expect(worker.discardOp('nonexistent-id')).resolves.toBeUndefined();
      });

      test('Test 6: retryOp emittiert status_change:syncing VOR syncAll', async () => {
        const storage = createInMemoryAdapter();
        const supabase = createSupabaseShim();
        const worker = new SyncWorker({ storage, supabase: supabase.client });
        const opId = await seedFailedOutboxEntry(storage);

        const events: SyncEvent[] = [];
        const unsub = syncEvents.on((e) => events.push(e));
        await worker.retryOp(opId);
        unsub();

        const statusChangeIdx = events.findIndex(
          (e) => e.type === 'status_change' && e.status === 'syncing',
        );
        const pushStartIdx = events.findIndex((e) => e.type === 'push_start');
        // status_change:syncing MUSS vor dem ersten push_start (oder alleine, falls outbox schon leer)
        expect(statusChangeIdx).toBeGreaterThanOrEqual(0);
        if (pushStartIdx >= 0) {
    ```

| Client-UI → local storage | User-Input (Retry/Discard-Buttons) löst direkte Storage-Operationen aus |
