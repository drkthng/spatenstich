```typescript
// und `.web.ts` für Web automatisch via Metro-Platform-Extensions.

export interface StrippedPhoto {
  /** Neue URI zum EXIF-freien Bild. Kann gleich der Input-URI sein (strippen in-place). */
  strippedUri: string;
  /** GPS-Koordinaten, nur wenn optIn=true UND Bild GPS enthält. Sonst null. */
  gps: { lat: number; lng: number } | null;
}

export interface StripExifOptions {
  /** DSGVO-Opt-in: wenn true, werden GPS-Koordinaten extrahiert (NICHT mitgesendet — separat persistiert). */
  optIn: boolean;

 * Plattform-Split via Metro-Extensions:
 *   exifStrip.native.ts → iOS/Android (react-native-exify + expo-image-manipulator)
 *   exifStrip.web.ts    → Web (piexifjs auf Blob/DataURL)

export declare function stripExifAndExtractGps(
  uri: string,
  options: StripExifOptions,
): Promise<StrippedPhoto>;
```

**3. `app/src/lib/photos/exifStrip.native.ts` (NEU):**

```typescript
import { readAsync } from '@lodev09/react-native-exify';
import { manipulateAsync, SaveFormat } from 'expo-image-manipulator';
import type { StrippedPhoto, StripExifOptions } from './exifStrip';

export async function stripExifAndExtractGps(
  uri: string,
  options: StripExifOptions,
): Promise<StrippedPhoto> {
  // 1. GPS extrahieren (nur bei Opt-in)
  let gps: { lat: number; lng: number } | null = null;
  if (options.optIn) {
    try {
      const exif = await readAsync(uri);
      if (exif && typeof exif.GPSLatitude === 'number' && typeof exif.GPSLongitude === 'number') {
        gps = { lat: exif.GPSLatitude, lng: exif.GPSLongitude };
      }
    } catch {
      // Kein EXIF oder nicht parsebar → kein GPS, harmlos
      gps = null;
    }
  }

  // 2. EXIF strippen via Re-Encode (garantiert EXIF-frei, unabhängig von EXIF-lib).
  //    expo-image-manipulator re-encoded das JPEG ohne EXIF-Segment.
  const result = await manipulateAsync(uri, [], {
    format: SaveFormat.JPEG,
    compress: 0.9,
    // expo-image-manipulator V53+ entfernt EXIF by default beim Re-Encode
  });

  return {
    strippedUri: result.uri,
    gps,
  };
}
```

**4. `app/src/lib/photos/exifStrip.web.ts` (NEU):**

```typescript
import piexif from 'piexifjs';
import type { StrippedPhoto, StripExifOptions } from './exifStrip';

// Web-Version: uri ist meist ein data:- oder blob:-URL vom <input type="file">
// oder vom ImagePicker (expo-image-picker auf Web).
// piexifjs arbeitet auf base64-DataURLs.

export async function stripExifAndExtractGps(
  uri: string,
  options: StripExifOptions,
): Promise<StrippedPhoto> {
  const dataUrl = await ensureDataUrl(uri);

  // 1. GPS extrahieren (nur bei Opt-in)
  let gps: { lat: number; lng: number } | null = null;
  if (options.optIn) {
    try {
      const exifObj = piexif.load(dataUrl);
      const gpsIfd = exifObj['GPS'] as Record<string, unknown> | undefined;
      if (gpsIfd) {
        const lat = parseGpsRational(gpsIfd[piexif.GPSIFD.GPSLatitude], gpsIfd[piexif.GPSIFD.GPSLatitudeRef]);
        const lng = parseGpsRational(gpsIfd[piexif.GPSIFD.GPSLongitude], gpsIfd[piexif.GPSIFD.GPSLongitudeRef]);
        if (lat !== null && lng !== null) gps = { lat, lng };
      }
    } catch {
      gps = null;
    }
  }

  // 2. EXIF strippen
  const stripped = piexif.remove(dataUrl);

  // 3. Als Blob-URL zurückgeben für Upload (ArrayBuffer-ready)
  const blob = await dataUrlToBlob(stripped);
  const strippedUri = URL.createObjectURL(blob);

  return { strippedUri, gps };
}

async function ensureDataUrl(uri: string): Promise<string> {
  if (uri.startsWith('data:')) return uri;
  // blob: oder http: URL → fetch + FileReader
  const response = await fetch(uri);
  const blob = await response.blob();
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.onload = () => resolve(reader.result as string);
    reader.onerror = reject;
    reader.readAsDataURL(blob);
  });
}

async function dataUrlToBlob(dataUrl: string): Promise<Blob> {
  const response = await fetch(dataUrl);
  return response.blob();
}

/**
 * piexifjs liefert GPS-Koordinaten als Rational-Arrays [[deg_num, deg_den], [min_num, min_den], [sec_num, sec_den]]
 * + Ref-String 'N'/'S'/'E'/'W'. Wir konvertieren zu Dezimalgrad.
 */
function parseGpsRational(
  raw: unknown,
  ref: unknown,
): number | null {
  if (!Array.isArray(raw) || raw.length !== 3) return null;
  try {
    const [d, m, s] = raw as Array<[number, number]>;
    const deg = d[0] / d[1];
    const min = m[0] / m[1];
    const sec = s[0] / s[1];
    let decimal = deg + min / 60 + sec / 3600;
    if (ref === 'S' || ref === 'W') decimal = -decimal;
    return decimal;
  } catch {
    return null;
  }
}
```

```typescript
import { storage } from '../../storage';
import { useAuthStore } from '../../stores/authStore';
import type { PhotoQueueRow } from '@spatenstich/shared';
import { stripExifAndExtractGps } from './exifStrip';
import { OutboxEnqueueError } from '../errors';
import { scheduleWriteDebounced } from '../sync/SyncTriggers';

function randomId(): string {
  if (typeof crypto !== 'undefined' && typeof crypto.randomUUID === 'function') {
    return crypto.randomUUID();
  }
  return `${Date.now()}-${Math.random().toString(36).slice(2, 10)}`;
}

/**
 * Fügt ein Foto zur lokalen Upload-Queue hinzu.
 * 1. EXIF-Strip (NFR-05)
 * 2. GPS extract (nur wenn optIn)
 * 3. writeWithOutbox(photo_queue, row, outbox)
 *
 * Die lokale URI bleibt in der Row als `storagePath`-Preview (vor Upload) — wird
 * beim Upload von PhotoUploader ersetzt durch den Remote-Pfad `<gardenId>/<photoId>.jpg`.
 *
 * Alternative wäre ein separates Feld `localUri` vs `remotePath`, aber die Row-Schema-
 * Definition (Plan 03-02) hat nur `storagePath`. Wir unterscheiden via `uploadStatus`:
 *   status='pending' → storagePath ist lokale URI
 *   status='uploaded' → storagePath ist Remote-Key
 */
export async function enqueuePhoto(
  gardenId: string,
  localUri: string,
  optIn: boolean,
): Promise<string> {
  const userId = useAuthStore.getState().userId;
  if (!userId) throw new Error('not_authenticated');

  // 1 + 2: EXIF strippen + GPS extrahieren (Task 03-05-01)
  const { strippedUri, gps } = await stripExifAndExtractGps(localUri, { optIn });

  const photoId = randomId();
  const now = new Date().toISOString();
  const row: PhotoQueueRow = {
    id: photoId,
    createdAt: now,
    updatedAt: now,
    updatedByUserId: userId,
    deletedAt: null,
    gardenId,
    storagePath: strippedUri, // lokale URI, wird beim Upload zu Remote-Key
    geoLat: gps?.lat ?? null,
    geoLng: gps?.lng ?? null,
    uploadStatus: 'pending',
    uploadError: null,
    jobId: null,
  };

  try {
    await storage.writeWithOutbox('photo_queue', row, {
      entity: 'photo_queue',
      rowId: photoId,
      operation: 'insert',
      payload: row,
    });
    scheduleWriteDebounced();
  } catch (cause) {
    throw new OutboxEnqueueError('photo_queue', photoId, cause);
  }
  return photoId;
}

export async function loadPendingPhotos(gardenId: string): Promise<PhotoQueueRow[]> {
  const rows = await storage.getRowsByGarden<PhotoQueueRow>('photo_queue', gardenId);
  return rows.filter((r) => r.uploadStatus === 'pending' || r.uploadStatus === 'failed');
}

export async function getPhoto(photoId: string): Promise<PhotoQueueRow | null> {
  return storage.getRow<PhotoQueueRow>('photo_queue', photoId);
}

/**
 * Patch eines photo_queue-Rows (z.B. upload_status update durch PhotoUploader).
 * Erzeugt Outbox-Eintrag (SyncWorker pusht den Patch später nach).
 */
export async function patchPhoto(
  photoId: string,
  patch: Partial<Pick<PhotoQueueRow, 'storagePath' | 'uploadStatus' | 'uploadError' | 'jobId'>>,
): Promise<void> {
  const userId = useAuthStore.getState().userId;
  if (!userId) throw new Error('not_authenticated');
  const existing = await storage.getRow<PhotoQueueRow>('photo_queue', photoId);
  if (!existing) throw new Error(`photo not found: ${photoId}`);
  const now = new Date().toISOString();
  const updated: PhotoQueueRow = {
    ...existing,
    ...patch,
    updatedAt: now,
    updatedByUserId: userId,
  };
  try {
    await storage.writeWithOutbox('photo_queue', updated, {
      entity: 'photo_queue',
      rowId: photoId,
      operation: 'update',
      payload: updated,
    });
    scheduleWriteDebounced();
  } catch (cause) {
    throw new OutboxEnqueueError('photo_queue', photoId, cause);
  }
}
```

```typescript
import * as Sentry from '@sentry/react-native';
import { supabase } from '../supabase';
import { storage } from '../../storage';
import { useAuthStore } from '../../stores/authStore';
import type { PhotoQueueRow } from '@spatenstich/shared';
import { patchPhoto } from './photoQueueRepo';
import { nextBackoffMs, MAX_ATTEMPTS } from '../sync/backoff';
import { syncEvents } from '../sync/events';

// Serialisation wie SyncWorker
let uploadInFlight = false;

const BUCKET = 'photos';

/**
 * Iteriert alle photo_queue-Rows mit upload_status='pending' (oder 'failed' < MAX_ATTEMPTS)
 * und versucht sie hochzuladen.
 */
export async function uploadPending(): Promise<void> {
  if (uploadInFlight) return;
  uploadInFlight = true;
  syncEvents.emit({ type: 'status_change', status: 'syncing' });
  try {
    const activeGardenId = useAuthStore.getState().activeGardenId;
    if (!activeGardenId) return;
    const allRows = await storage.getRowsByGarden<PhotoQueueRow>('photo_queue', activeGardenId);
    const pending = allRows.filter((r) => r.uploadStatus === 'pending' || r.uploadStatus === 'failed');
    for (const row of pending) {
      await uploadOne(row);
    }
  } finally {
    uploadInFlight = false;
  }
}

async function uploadOne(row: PhotoQueueRow): Promise<void> {
  try {
    // 1. URI → ArrayBuffer (L-7: iOS 0-byte-Blob-Bug → fetch+arrayBuffer statt FormData)
    const response = await fetch(row.storagePath);
    if (!response.ok) throw new Error(`fetch ${row.storagePath}: ${response.status}`);
    const buffer = await response.arrayBuffer();
    if (buffer.byteLength === 0) throw new Error('empty_blob');

    // 2. Upload nach Supabase Storage (Bucket 'photos', RLS in Plan 03-01 geregelt)
    const remotePath = `${row.gardenId}/${row.id}.jpg`;
    const { error: uploadErr } = await supabase.storage
      .from(BUCKET)
      .upload(remotePath, buffer, {
        contentType: 'image/jpeg',
        upsert: false,
        cacheControl: '3600',
      });
    if (uploadErr) throw uploadErr;

    // 3. RPC enqueue_photo_analysis (Plan 03-01 Section 7)
    // Vorher: photo_queue-Row muss auf Server existieren. Der SyncWorker pushed parallel.
    // Falls RPC mit "row not found" failed: patchen + Retry beim nächsten Reconnect.
    const { data: jobId, error: rpcErr } = await supabase.rpc('enqueue_photo_analysis', {
      p_photo_id: row.id,
    });
    if (rpcErr) throw rpcErr;

    // 4. Row lokal patchen: status=uploaded, storagePath=remote, jobId
    await patchPhoto(row.id, {
      storagePath: remotePath,
      uploadStatus: 'uploaded',
      uploadError: null,
      jobId: jobId as string,
    });

    Sentry.addBreadcrumb({
      category: 'photo_upload',
      level: 'info',
      message: `photo uploaded ${row.id}`,
      data: { row_id: row.id, garden_id: row.gardenId, job_id: jobId },
    });
  } catch (e) {
    await handleUploadError(row, e);
  }
}

async function handleUploadError(row: PhotoQueueRow, e: unknown): Promise<void> {
  const err = e as { code?: string; message?: string };
  const msg = err.message ?? String(e);
  const nextAttempts = (row.updatedAt ? 1 : 0) + 1; // row.attempts is NOT on row; we track via photo_queue reattempt

  // photo_queue.upload_status='failed' + upload_error; Outbox-Eintrag via patchPhoto
  if (nextAttempts >= MAX_ATTEMPTS) {
    await patchPhoto(row.id, {
      uploadStatus: 'failed',
      uploadError: `MAX_ATTEMPTS: ${msg}`,
    });
    Sentry.captureException(e, {
      tags: { sync_phase: 'photo_upload', error_kind: 'max_attempts', row_id: row.id },
    });
    return;
  }

  await patchPhoto(row.id, {
    uploadStatus: 'failed',
    uploadError: msg,
  });
  Sentry.captureException(e, {
    tags: { sync_phase: 'photo_upload', row_id: row.id },
  });
}

export function _resetPhotoUploader(): void {
  uploadInFlight = false;
}
```
