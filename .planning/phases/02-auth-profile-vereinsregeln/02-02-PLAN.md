**D. `app/src/lib/profileRepo.ts`** — mode-aware persistence:
```typescript
import { supabase } from './supabase';
import { storage } from '../storage';
import { useAuthStore } from '../stores/authStore';
import type { UserProfile } from '@spatenstich/shared';

const PROFILE_KEY = 'profile';

export async function loadProfile(): Promise<Partial<UserProfile> | null> {
  const { mode, userId } = useAuthStore.getState();
  if (!mode || !userId) return null;
  if (mode === 'account') {
    const { data, error } = await supabase.from('profiles').select('*').eq('id', userId).maybeSingle();
    if (error) throw error;
    return data ? { plz: data.plz, klimazone: data.klimazone as any, archetype: data.archetype as any } : null;
  } else {
    const raw = await storage.get(PROFILE_KEY);
    return raw ? JSON.parse(raw) : null;
  }
}

export async function saveProfile(patch: Partial<UserProfile>): Promise<void> {
  const { mode, userId } = useAuthStore.getState();
  if (!mode || !userId) throw new Error('Not authenticated');
  if (mode === 'account') {
    const { error } = await supabase.from('profiles').upsert({ id: userId, ...patch });
    if (error) throw error;
  } else {
    const current = await loadProfile();
    const merged = { ...(current ?? {}), ...patch };
    await storage.set(PROFILE_KEY, JSON.stringify(merged));
  }
}
```

**E. `app/src/hooks/useProfile.ts`** — wraps profileStore + repo:
```typescript
import { useEffect } from 'react';
import { useProfileStore } from '../stores/profileStore';
import { loadProfile, saveProfile } from '../lib/profileRepo';

export function useProfile() {
  const store = useProfileStore();
  useEffect(() => {
    loadProfile().then((p) => {
      if (p?.plz && p?.klimazone) store.setPlz(p.plz, p.klimazone);
      if (p?.archetype) store.setArchetype(p.archetype);
    }).catch(() => { /* silent — banners will show missing data */ });
  // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);

  return {
    plz: store.plz,
    klimazone: store.klimazone,
    archetype: store.archetype,
    vereinsregeln: store.vereinsregeln,
    setPlz: async (plz: string, klimazone: any) => { store.setPlz(plz, klimazone); await saveProfile({ plz, klimazone }); },
    setArchetype: async (archetype: any) => { store.setArchetype(archetype); await saveProfile({ archetype }); },
  };
}
```
