**Step 3: Implement `app/src/lib/importValidator.ts`:**

```typescript
import Ajv2020 from 'ajv/dist/2020';       // NOT default import — Pitfall from RESEARCH
import addFormats from 'ajv-formats';
import schema from '../../../schemas/spatenstich-import.v1.json';
import type { ImportPayload } from '@spatenstich/shared';

const ajv = new Ajv2020({ allErrors: true });
addFormats(ajv);
const validate = ajv.compile(schema);

export type ValidationResult =
  | { ok: true; payload: ImportPayload }
  | { ok: false; errors: string[] };

export function validatePayload(raw: unknown): ValidationResult {
  const valid = validate(raw);
  if (!valid) {
    const errors = (validate.errors ?? []).map((e) =>
      `${e.instancePath || '(root)'}: ${e.message}`
    );
    return { ok: false, errors };
  }
  // Cross-reference check: plants.bedRef must exist in beds (per D-12)
  const payload = raw as ImportPayload;
  const bedIds = new Set((payload.beds ?? []).map((b) => b.localId));
  const crossRefErrors: string[] = [];
  for (const plant of payload.plants ?? []) {
    if (plant.bedRef && !bedIds.has(plant.bedRef)) {
      crossRefErrors.push(`plant "${plant.localId}": unbekannte bedRef "${plant.bedRef}"`);
    }
  }
  if (crossRefErrors.length > 0) {
    return { ok: false, errors: crossRefErrors };
  }
  return { ok: true, payload };
}
```

IMPORTANT: If `ajv/dist/2020` import fails under Metro/Jest, try `const Ajv2020 = require('ajv/dist/2020')` as fallback. Check that `resolveJsonModule: true` exists in the tsconfig used by the app workspace — if missing, add it.
